function sum(){
	if(arguments.length == 2){
		return arguments[0]+arguments[1];
	}

	var num1 = arguments[0];
	return function(num2){
		return num1+num2;
	}
}

console.log(sum(2,3));   //5
console.log(sum(2)(3));   //5
