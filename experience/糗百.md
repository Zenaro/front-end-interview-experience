用c语言实现求一个字符串长度的函数，不用中间变量
int My_Strlen(const char *ps)//不用中间变量 用递归实现	，在字符串中通常可以利用最后一个结束符’\0’
{
    if((ps == NULL)||(*ps == '\0'))
    {
        return 0;  
    } 
    else
    {
        return 1+My_Strlen(++ps);
    }
}
int My_Strlen(const char *ps)//常规实现 声明一个整形变量
{
    int len = 0;
    if(ps == NULL)
    {
        return 0;
    } 
    for(;*ps++ != '\0';)
    {
        len++; 
    }
   
    return 0;
   
}
